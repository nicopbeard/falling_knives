DELETE FROM market WHERE dataid, date, high, low, open, close, volume IN (SELECT dataid, date, high, low, open, close, volume FROM  (SELECT dataid, date, high, low, open, close, volume, ROW_NUMBER() OVER( PARTITION BY dataid, date, high, low, open, close, volume ORDER BY  dataid, date, high, low, open, close, volume) AS row_num FROM market) t WHERE t.row_num > 1 );
